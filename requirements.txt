# NOTE: ROS2 Dependencies
# The following are typically installed via ROS2 system packages (apt):
# - rclpy (ROS2 Python client library)
# - gazebo_msgs (Gazebo ROS2 message definitions)
# - geometry_msgs (Geometry message definitions)
# - sensor_msgs (Sensor message definitions)
# - std_msgs (Standard message definitions)
# - std_srvs (Standard service definitions)
# - nav_msgs (Navigation message definitions)
#
# Install ROS2 Humble or Jazzy via:
#   sudo apt install ros-humble-desktop ros-humble-gazebo-ros-pkgs
# or
#   sudo apt install ros-jazzy-desktop ros-jazzy-gazebo-ros-pkgs
#
# Then source the ROS2 workspace:
#   source /opt/ros/humble/setup.bash  # or jazzy

# NOTE: Gazebo Dependencies

#   sudo apt install gz-harmonic  # or gz-garden
#
